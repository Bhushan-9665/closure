    Que-4
        Write a code to calculate area of a rectangle using inner function. In this case outer function should accept 
        parameter length and inner function should accept parameter breadth.


    Output-
        function calculateArea(length) {
            function calculateInnerArea(breadth) {
            return length * breadth;
            }
        
            return calculateInnerArea;
        }
        
        const length = 5;
        const breadth = 8;
        const areaCalculator = calculateArea(length);
        const area = areaCalculator(breadth);
